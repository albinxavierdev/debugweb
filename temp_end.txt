      function scrollRight() {
          if (currentIndex < maxIndex) {
              currentIndex++;
              updateCarouselPosition();
          }
          updateButtonStates();
          pauseAllVideos();
      }
      
      function updateCarouselPosition() {
          const scrollAmount = currentIndex * itemWidth;
          carousel.style.transform = `translateX(-${scrollAmount}px)`;
      }
      
      function updateButtonStates() {
          // Disable left button at the beginning
          leftBtn.disabled = currentIndex === 0;
          leftBtn.style.opacity = currentIndex === 0 ? '0.5' : '1';
          
          // Disable right button at the end
          rightBtn.disabled = currentIndex >= maxIndex;
          rightBtn.style.opacity = currentIndex >= maxIndex ? '0.5' : '1';
      }
      
      // Function to pause all videos when scrolling
      function pauseAllVideos() {
          const iframes = document.querySelectorAll('iframe');
          iframes.forEach(iframe => {
              // Save the current src
              const src = iframe.src;
              // Reset the src to pause the video
              iframe.src = src;
          });
      }

      // Auto-adjust height for mobile
      function adjustForMobile() {
          if (window.innerWidth <= 768) {
              document.querySelectorAll('.video-container.shorts').forEach(container => {
                  container.style.paddingTop = '100%'; // Square format on mobile
              });
          } else {
              document.querySelectorAll('.video-container.shorts').forEach(container => {
                  container.style.paddingTop = '177.78%'; // 9:16 on desktop
              });
          }
      }
      
      // Call on load
      adjustForMobile();
      
      // Add click handlers to make thumbnails open YouTube videos in a new tab
      document.querySelectorAll('.carousel-item.shorts-item').forEach(function(item) {
          item.style.cursor = 'pointer';
          item.addEventListener('click', function(e) {
              // Prevent default behavior and stop propagation
              e.preventDefault();
              e.stopPropagation();
              
              // Get the iframe within this item
              const iframe = item.querySelector('iframe');
              if (iframe) {
                  // Extract video ID from src
                  const src = iframe.getAttribute('src');
                  const videoId = src.split('?')[0].split('/').pop();
                  
                  // Open as a YouTube short in a new tab
                  window.open(`https://www.youtube.com/shorts/${videoId}`, '_blank');
              }
          });
      });

      // Add click handlers to play videos in-place
      document.querySelectorAll('.carousel-item.shorts-item').forEach(function(item) {
          item.style.cursor = 'pointer';
          
          // Create reset button for each video
          const resetButton = document.createElement('div');
          resetButton.className = 'video-reset-button';
          resetButton.innerHTML = 'âœ•';
          resetButton.addEventListener('click', function(e) {
              e.stopPropagation(); // Prevent triggering the parent click
              resetVideo(item);
          });
          item.appendChild(resetButton);
          
          item.addEventListener('click', function(e) {
              // Prevent default behavior and stop propagation
              e.preventDefault();
              e.stopPropagation();
              
              // If another video is playing, reset it first
              document.querySelectorAll('.carousel-item.shorts-item.playing').forEach(function(playingItem) {
                  if (playingItem !== item) {
                      resetVideo(playingItem);
                  }
              });
              
              // Get the iframe within this item
              const iframe = item.querySelector('iframe');
              if (iframe) {
                  // Get current src and modify to enable autoplay and show controls
                  let src = iframe.getAttribute('src');
                  
                  // If the video is not already playing
                  if (!item.classList.contains('playing')) {
                      // Update src to autoplay with controls
                      src = src.replace('controls=0', 'controls=1');
                      if (!src.includes('autoplay=1')) {
                          src = src + '&autoplay=1';
                      }
                      iframe.setAttribute('src', src);
                      
                      // Update styles for playing state
                      iframe.style.top = '0';
                      iframe.style.height = '100%';
                      iframe.style.opacity = '1';
                      iframe.style.pointerEvents = 'auto';
                      
                      // Mark as playing
                      item.classList.add('playing');
                      
                      // Pause auto-scrolling while video is playing
                      clearInterval(autoScrollInterval);
                  }
              }
          });
      });
  });

  // Function to reset a video to thumbnail state
  function resetVideo(item) {
      const iframe = item.querySelector('iframe');
      if (iframe) {
          // Get the current src
          let src = iframe.getAttribute('src');
          
          // Reset controls and remove autoplay
          src = src.replace('controls=1', 'controls=0');
          src = src.replace('&autoplay=1', '');
          
          // Reset src to stop video
          iframe.setAttribute('src', src);
          
          // Reset styles
          iframe.style.top = '-60%';
          iframe.style.height = '250%';
          iframe.style.opacity = '0.99';
          iframe.style.pointerEvents = 'none';
          
          // Remove playing class
          item.classList.remove('playing');
          
          // Restart auto-scrolling
          autoScrollInterval = setInterval(autoScroll, 6000);
      }
  }
